### Install
```
npm install --save isihac
```


### Example
```
var isihac = require('isihac');

var allQuotes = isihac.allQuotes(); // retrieve all quotes
var randomQuote = isihac.randomQuote(); // retrieve a random isihac quote
```

### Run tests
```
npm run tests
```
